\begin{theorem}\label{thm:main}
For all $n$, $\sum_{k=1}^n k = \frac{n(n+1)}{2}$.
\end{theorem}
\begin{proof}
By induction on $n$.
\end{proof}
\begin{lemma}
Trivial.
\end{lemma}
Theorem~\ref{thm:main} is standard.
