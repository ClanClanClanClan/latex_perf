\section{有限元离散}

选取三角形单元，在每个单元$K$上定义分片线性函数空间
\[
  V_h = \{v_h \in C(\bar{\Omega}) : v_h|_K \in P_1(K), \forall K \in \mathcal{T}_h\}.
\]

\begin{theorem}
设$u \in H^2(\Omega)$为精确解，$u_h \in V_h$为有限元解，则
\[
  \|u - u_h\|_{H^1(\Omega)} \le Ch\|u\|_{H^2(\Omega)},
\]
其中$C$为与$h$无关的常数。
\end{theorem}
